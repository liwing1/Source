36/* Several timeslots may be required through the day, to allow for lunchtime,
37 early evening, etc.
38 Three types of day are usually enough - weekdays, Saturday and Sunday/public
39 holidays.
40 The maximum number of actual tariffs is generally only 2 or 3 - something like
41 normal, peak and overnight cheap rate
42*/
43
44/*! These structures are images of entries in the EEPROM */
45/*! This structure contains the information about a daily tariff schedule. It is an image
46 of the data stored in external EEPROM */
47typedef struct
48{
49 /*! \brief The tariff type */
50 uint8_t tariff:3;
51 /*! \brief The starting hour of the day (0-23). */
52 uint8_t start_hour:5;
53 /*! \brief The starting minute of the hour (0-59). */
54 uint8_t start_minute;
55} eeprom_day_schedule_timeslot_t;
56
57/*! This structure contains the information about a single public holiday. It is an image
58 of the data stored in external EEPROM */
59typedef struct
60{
61 /*! \brief The year of the holiday (0 = 2000, to 255 = 2255). */
62 uint8_t year;
63 /*! \brief The month of the holiday (1-12). */
64 uint8_t month:4;
65 /*! \brief Spare bits */
66 uint8_t spare:4;
67 /*! \brief The day of the month of the holiday (1-31). */
68 uint8_t day:5;
69 /*! \brief The tariff schedule type for the day. */
70 uint8_t day_type:3;
71} eeprom_holiday_t;
72
73/*! This structure contains the information about a single billing cutoff date. It is an image
74 of the data stored in external EEPROM */
75typedef struct
76{
77 /*! \brief Billing cutoff year (0 = 2000, to 255 = 2255). */
78 uint8_t year;
79 /*! \brief Billing cutoff month (1-12). */
80 uint8_t month:4;
81 /*! \brief Spare bits */
82 uint8_t spare:4;
83 /*! \brief Billing cutoff day of the month (1-31). */
84 uint8_t day;
85} eeprom_cutoff_date_t;

87/*! This structure contains the information about a daily peak usage record. It is an image
88 of the data stored in external EEPROM */
89typedef struct
90{
91 /*! \brief The peak 32 bit energy */
92 uint32_t usage;
93 /*! \brief The hour of the peak (0-23). */
94 uint8_t hour;
95 /*! \brief The minute of the hour of the peak (0-59). */
96 uint8_t minute;

99/*! This structure contains the information about a usage record. It is an image of the data
100 stored in external EEPROM */
101typedef struct

103 /*! \brief High 32 bits of the 64 bit energy */
104 uint32_t energy_hi;

108 uint8_t sumcheck;

111/* Limits to size the various sections in the EEPROM */
112
113#define MULTIRATE_DAY_SCHEDULES 8
114#define MULTIRATE_DAY_SCHEDULE_TIMESLOTS 8
115#define MULTIRATE_MAX_HOLIDAYS 32
116#define MULTIRATE_MAX_CUTOFF_DATES 12
117#define MULTIRATE_MAX_DAILY_PEAKS 31
118
119/* Positions of the various data sections in the EEPROM */
120
121/* A signature to identify the EEPROM status */
122#define EEPROM_START_SIGNATURE 0
123/* Tariff schedule */
124#define EEPROM_START_DAY_SCHEDULES 4
125/* Weekday tariff schedule pattern */
126#define EEPROM_START_WEEKDAYS (EEPROM_START_DAY_SCHEDULES + MULTIRATE_DAY_SCHEDULES*MULTIRATE_DAY_SCHEDULE_TIMESLOTS*sizeof(eeprom_day_schedule_timeslot_t))
127/* Holidays, where special tariff schedules apply */
128#define EEPROM_START_HOLIDAYS (EEPROM_START_WEEKDAYS + 4)
129/* Cutoff days */
130#define EEPROM_START_CUTOFF_DATES (EEPROM_START_HOLIDAYS + MULTIRATE_MAX_HOLIDAYS*sizeof(eeprom_holiday_t))
131/* Daily usage peaks */
132#define EEPROM_START_PEAKS (EEPROM_START_CUTOFF_DATES + MULTIRATE_MAX_CUTOFF_DATES*sizeof(eeprom_cutoff_date_t))
133/* Usage history entries */
134#define EEPROM_START_HISTORIES_A (EEPROM_START_PEAKS + MULTIRATE_MAX_DAILY_PEAKS*sizeof(eeprom_daily_peak_t))
135#define EEPROM_START_HISTORIES_B (EEPROM_START_HISTORIES_A + MULTIRATE_MAX_CUTOFF_DATES*MULTIRATE_DAY_SCHEDULES*sizeof(eeprom_history_t))
136#define EEPROM_START_HISTORIES_C (EEPROM_START_HISTORIES_B + MULTIRATE_MAX_CUTOFF_DATES*MULTIRATE_DAY_SCHEDULES*sizeof(eeprom_history_t))
137/* End of EEPROM */
138#define EEPROM_END (EEPROM_START_HISTORIES_C + MULTIRATE_MAX_CUTOFF_DATES*MULTIRATE_DAY_SCHEDULES*sizeof(eeprom_history_t))
